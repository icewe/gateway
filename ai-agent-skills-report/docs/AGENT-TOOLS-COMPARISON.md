```typescript
// Cursor Context 收集
class ContextManager {
  async collectContext(): Promise<Context> {
    const [openFiles, terminal, problems, search] = await Promise.all([
      this.getOpenFiles(),
      this.getTerminalHistory(),
      this.getProblems(),
      this.getRecentSearch()
    ]);
    
    return {
      files: openFiles,
      terminal,
      diagnostics: problems,
      search
    };
  }
  
  // Tab 文件追踪
  async trackTabs(): Promise<TabContext> {
    return {
      active: this.getActiveTab(),
      recent: this.getRecentTabs(10),
      pinned: this.getPinnedTabs()
    };
  }
}
```

### 核心特点

```typescript
// Windsurf Flow
class FlowEngine {
  async executeFlow(flow: Flow, context: Context): Promise<void> {
    for (const step of flow.steps) {
      // 执行每一步
      await this.executeStep(step, context);
      
      // 验证结果
      if (!this.validateStep(step, context)) {
        throw new Error(`Flow failed at step: ${step.name}`);
      }
    }
  }
  
  // 步骤类型
  stepTypes = {
    edit: EditStep,
    create: CreateStep,
    delete: DeleteStep,
    command: CommandStep,
    agent: AgentStep
  };
}
```

## 对比总结

| 特性 | Claude Code | Cursor | Windsurf |
|------|-------------|--------|----------|
| **架构** | CLI + MCP | VSCode 插件 | VSCode 插件 |
| **交互** | 纯对话 | 对话 + Cmd+K | 对话 + Flows |
| **编辑** | 工具调用 | 实时编辑 | 流程自动化 |
| **上下文** | 项目级 | 文件级 | 多文件 |
| **扩展性** | MCP | 插件 | 自定义 |
